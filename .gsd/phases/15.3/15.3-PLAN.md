         try { await client.query("ROLLBACK"); } catch (_) {}

    class Money {
      static toCents(amount) {
        if (amount === null || amount === undefined) throw new Error('Money.toCents: amount is required');
        if (typeof amount === 'string') amount = parseFloat(amount);
        if (isNaN(amount) || amount < 0) throw new Error(`Money.toCents: invalid amount "${amount}"`);
        return Math.round(amount * 100);
      }
      static fromCents(cents) { return cents / 100; }
      static format(amount, currency = 'KES') { return `${currency} ${parseFloat(amount).toFixed(2)}`; }
    }
    module.exports = Money;
